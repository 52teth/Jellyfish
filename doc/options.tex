\subsection{count}

Count k-mers or qmers in fasta or fastq files

Usage: jellyfish count [OPTIONS]... [file.f[aq]]...

\begin{description}
\item[\Opt{-h},\LOpt{help}] Print help and exit
\item[\LOpt{full-help}] Print help, including hidden options, and exit
\item[\Opt{-V},\LOpt{version}] Print version and exit
\item[\Opt{-m},\LOptArg{mer-len}{=INT}] Length of mer (mandatory)
\item[\Opt{-s},\LOptArg{size}{=LONG}] Hash size (mandatory)
\item[\Opt{-t},\LOptArg{threads}{=INT}] Number of threads  (default=`1')
\item[\Opt{-o},\LOptArg{output}{=STRING}] Output prefix  (default=`mer\_counts')
\item[\Opt{-c},\LOptArg{counter-len}{=Length}] in bits
Length of counting field  (default=`7')
\item[\LOptArg{out-counter-len}{=Length}] in bytes
Length of counter field in output  
(default=`4')
\item[\Opt{-C},\LOpt{both-strands}] Count both strand, canonical representation  
(default=off)
\item[\Opt{-p},\LOptArg{reprobes}{=INT}] Maximum number of reprobes  (default=`62')
\item[\Opt{-r},\LOpt{raw}] Write raw database  (default=off)
\item[\Opt{-q},\LOpt{quake}] Quake compatibility mode  (default=off)
\item[\LOptArg{quality-start}{=INT}] Starting ASCII for quality values  
(default=`64')
\item[\LOptArg{min-quality}{=INT}] Minimum quality. A base with lesser quality 
becomes an N  (default=`0')
\item[\Opt{-L},\LOptArg{lower-count}{=LONG}] Don't output k-mer with count < lower-count
\item[\Opt{-U},\LOptArg{upper-count}{=LONG}] Don't output k-mer with count > upper-count
\item[\LOptArg{matrix}{=Matrix}] file      Hash function binary matrix
\item[\LOptArg{timing}{=Timing}] file      Print timing information
\end{description}
